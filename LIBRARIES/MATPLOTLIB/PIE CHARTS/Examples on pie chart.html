#Adding autopct attribute
from matplotlib import pyplot as plt
plt.title("Games View",color="green",fontweight="bold",fontsize=20)
sports=[10,5,5,10,10]
spnames=["FootBall","Hockey","Cricket","Basketball","Badminton"]
cols=["gold","green","hotpink","cyan","blue"]
plt.pie(sports,labels=spnames,explode=[0.1,0,0,0,0],colors=cols,startangle=90,autopct='%0.0f%%',
        wedgeprops={'linewidth':4,'edgecolor':"red"})
plt.show()

#Question: Priya lists down her monthly expenditure as follows:

#			Expenditure	Amount
#			Rent		4000
#			Food		5400

#			Clothing	2800
#			Savings		400
#		Draw a pie chart for her monthly expenses.
#Adding autopct attribute
                
from matplotlib import pyplot as plt
plt.title("Priya Expenditure",color="green",fontweight="bold",fontsize=20)
expen=[4000,5400,2800,400]
expnames=["Rent","Food","Clothing","Savings"]
cols=["gold","green","hotpink","cyan"]
plt.pie(expen,labels=expnames,colors=cols,autopct='%0.00f%%',
        wedgeprops={'linewidth':4,'edgecolor':"red"})
plt.legend()
plt.show()
